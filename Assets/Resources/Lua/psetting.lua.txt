
function awake()
	local setBtnEvent = CS.PUtils.setBtnEvent
	local tm = self.transform
	local nm = CS.NetMgr.GetInstance()

	setBtnEvent(tm, "Bottom/BtnLogout", function()
		nm:logout()
	end)
end

function enter()
	self:show()

	local setText = CS.PUtils.setText

	setText(self.transform, "items/grid_ign/version/version", "V"..CS.GameSettings.Instance.version)

	self:Reposition("items/grid_ign")
end
